### Use Memo
---
```jsx
import { useMemo } from 'react';

function Typeahead({users}) {
  const filteredusers = useMemo(
    () => filterUsers(users, searchText),
    [users, searchText]
  );
  
  return (
    <>
      <UserList users={filteredUsers}/>
    </>
  );
}
```
